ai-proxy-server.ts

import ZAI from "z-ai-web-dev-sdk";

const PORT = 8080;

// Internal AI Gateway URL (handles authentication to real Z.ai API)
const GATEWAY_URL = "http://172.25.136.193:8080/v1";
const GATEWAY_API_KEY = "Z.ai";

console.log(`🚀 AI Proxy Server starting on port ${PORT}...`);
console.log(`📡 Gateway URL: ${GATEWAY_URL}`);

// Create a simple proxy server using Bun's built-in server
Bun.serve({
    port: PORT,
    async fetch(request) {
        const url = new URL(request.url);
        const path = url.pathname;
        
        // CORS headers - IMPORTANT: Allows cross-origin requests
        const corsHeaders = {
            "Access-Control-Allow-Origin": "*",
            "Access-Control-Allow-Methods": "GET, POST, OPTIONS",
            "Access-Control-Allow-Headers": "Content-Type, Authorization",
        };
        
        // Handle preflight requests
        if (request.method === "OPTIONS") {
            return new Response(null, { headers: corsHeaders });
        }
        
        try {
            const zai = await ZAI.create();
            
            // Chat completions endpoint
            if (path === "/v1/chat/completions" && request.method === "POST") {
                const body = await request.json();
                console.log("[Proxy] Chat request:", body.messages?.length, "messages");
                
                const result = await zai.chat.completions.create(body);
                return Response.json(result, { headers: corsHeaders });
            }
            
            // Image generation endpoint
            if (path === "/v1/images/generations" && request.method === "POST") {
                const body = await request.json();
                console.log("[Proxy] Image request:", body.prompt?.substring(0, 50));
                
                const result = await zai.images.generations.create(body);
                return Response.json(result, { headers: corsHeaders });
            }
            
            // Video generation endpoint
            if (path === "/v1/video/generations" && request.method === "POST") {
                const body = await request.json();
                console.log("[Proxy] Video request:", body.prompt?.substring(0, 50));
                
                const result = await zai.video.generations.create(body);
                return Response.json(result, { headers: corsHeaders });
            }
            
            // Video status check endpoint
            // Uses internal gateway's /async-result endpoint with proper headers
            if (path.startsWith("/v1/video/generations/") && request.method === "GET") {
                const taskId = path.split("/").pop();
                console.log("[Proxy] Video status check:", taskId);
                
                // Use gateway's async-result endpoint (the SDK uses this format too)
                const statusUrl = `${GATEWAY_URL}/async-result?id=${encodeURIComponent(taskId)}`;
                console.log("[Proxy] Fetching:", statusUrl);
                
                const response = await fetch(statusUrl, {
                    method: 'GET',
                    headers: {
                        'Authorization': `Bearer ${GATEWAY_API_KEY}`,
                        'Content-Type': 'application/json',
                        'X-Z-AI-From': 'Z'  // Required header for gateway authentication
                    }
                });
                if (!response.ok) {
                    const errorText = await response.text();
                    console.error("[Proxy] Video status error:", response.status, errorText);
                    throw new Error(`Video status check failed: ${errorText}`);
                }
                const result = await response.json();
                console.log("[Proxy] Video status result:", JSON.stringify(result).substring(0, 200));
                return Response.json(result, { headers: corsHeaders });
            }
            
            // Function invoke endpoint (web search, etc.)
            if (path === "/v1/functions/invoke" && request.method === "POST") {
                const body = await request.json();
                console.log("[Proxy] Function invoke:", body.name);
                
                const result = await zai.functions.invoke(body.name, body.arguments);
                return Response.json(result, { headers: corsHeaders });
            }
            
            // Health check endpoint
            if (path === "/health") {
                return Response.json({ 
                    status: "ok", 
                    message: "AI Proxy Server running",
                    timestamp: new Date().toISOString()
                }, { headers: corsHeaders });
            }
            
            // Root endpoint - shows available endpoints
            if (path === "/") {
                return Response.json({
                    service: "OpenMosaic AI Proxy",
                    endpoints: [
                        "POST /v1/chat/completions",
                        "POST /v1/images/generations", 
                        "POST /v1/video/generations",
                        "GET /v1/video/generations/:taskId",
                        "POST /v1/functions/invoke",
                        "GET /health"
                    ]
                }, { headers: corsHeaders });
            }
            
            return Response.json({ error: "Not found" }, { status: 404, headers: corsHeaders });
            
        } catch (error) {
            console.error("[Proxy] Error:", error);
            return Response.json({ 
                error: error instanceof Error ? error.message : "Unknown error" 
            }, { status: 500, headers: corsHeaders });
        }
    }
});

console.log(`✅ AI Proxy Server running on http://localhost:${PORT}`);
console.log(`📡 Endpoints available:`);
console.log(`   POST /v1/chat/completions`);
console.log(`   POST /v1/images/generations`);
console.log(`   POST /v1/video/generations`);
console.log(`   GET  /health`);
